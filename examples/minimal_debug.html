# Helper to convert a 1d data list to a 2d list
def to_matrix(l, n):
    return [l[i:i+n] for i in range(0, len(l), n)]

data = [x**2 for x in range(0, 25)]
fig = go.Figure(data=go.Heatmap(
    z=to_matrix(data, 5),
    x=list(range(0, 5)),
    y=list(range(0, 5)))
)

fig.update_layout(
    title='Plot'
)

fig.show(renderer="notebook_connected")
